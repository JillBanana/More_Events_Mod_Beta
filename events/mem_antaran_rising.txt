namespace = mem_antaran_test

country_event = {
	id = mem_antaran_test.1
	hide_window = yes
	
	is_triggered_only = yes
	
	immediate = {
		random_system = {
			limit = {
				is_rim_system = yes
			}
			spawn_system = { 
				min_distance = 70 
				max_distance = 150
				initializer = "mem_antaran_homesystem"
				hyperlane = no 
				is_discovered = yes
			}
		}
	}
}

country_event = {
	id = mem_antaran_test.2
	hide_window = yes
	is_triggered_only = yes

	immediate = {
		random_system = {
			limit = {
				has_owner = yes
				starting_system = yes
				owner = {
					is_same_value = root
				}
			}
			random_system = {
				limit = {
					has_star_flag = mem_antaran_homesystem
				}
				prev = { # done this way to ensure we don't spawn "orphan" wormholes (wormholes that aren't linked to anything)
						spawn_natural_wormhole = {
						random_pos = yes
						orbit_angle = 360
					}
				}
				spawn_natural_wormhole = {
					random_pos = yes
					orbit_angle = 360
				}
				link_wormholes = prev
			}
		}
		add_seen_bypass_type = wormhole
	}
}
