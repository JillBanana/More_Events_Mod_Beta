@time_limit = 2

#Timer loop
planet_event = {
    id = mem_rubicon.3
    hide_window = yes
    is_triggered_only = yes

    trigger = {
        NOT = {
            has_global_flag = mem_rubicon_completed
        }
    }

    immediate = {
        IF = {
            limit = {
                has_research_station = yes
                exists = space_owner
                space_owner = {
                    check_variable = {
                        which = mem_rubicon_timer
                        value < @time_limit
                    }
                }
            }
            space_owner = {
                change_variable = {
                    which = mem_rubicon_timer
                    value = 1
                }
                IF = {
                    limit = {
                        check_variable = {
                            which = mem_rubicon_timer
                            value = @time_limit
                        }
                        NOT = {
                            has_country_flag = mem_rubicon_gained_news
                        }
                    }
                    country_event = {
                        id = mem_rubicon.4
                    }
                }
            }
        }
        planet_event = { id = mem_rubicon.3 days = 2 random = 70 }
    }
}
