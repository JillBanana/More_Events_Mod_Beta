
# Lost Emperor, any star except black holes
mem_lost_emperor_category = {
	picture = "GFX_evt_star_yellow"
	level = 4
	
	spawn_chance = {		
		modifier = {
			add = 2
			is_star = yes
			NOT = { is_planet_class = pc_black_hole }
			NOT = { is_planet_class = pc_pulsar }

			NOT = {
				solar_system = {
					OR = {
						any_planet = {
							OR = {
								is_colonizable = yes
								is_colony = yes
							}
						}
						has_star_flag = enclave
					}
				}
			}
		}
	}
	
	on_success = mem_lost_emperor.1
}
