	key = mem_refuel_category
	desc = mem_refuel_category_desc
	picture = GFX_evt_gas_giant

	spawn_chance = {
		modifier = {
			is_planet_class = pc_gas_giant
			from.owner = {
				NOT = {
					has_country_flag = mem_refuel_anomaly_spawned
				}
			}
			add = 3
		}

			set_country_flag = mem_refuel_anomaly_spawned
		}
	}

		ship_event = {
			id = mem_refuel.20
		}
	}

	on_critical_fail = {
		ship_event = {
			id = mem_refuel.30
		}
	}
}
