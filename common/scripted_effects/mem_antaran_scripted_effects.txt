# Checks if Ancient_Protectors country (global event target) exists
# Creates the country and saves it if it does not

create_mem_antaran_country = {
	if = {
		limit = {
			NOT = {
				exists = event_target:mem_orila_country
			}
		}
		create_species = {
			name = "Antarans"
			class = ART
			portrait = art17
			traits = {
				trait = trait_intelligent
				trait = trait_natural_physicists
				trait = trait_venerable
				trait = trait_slow_breeders
				trait = trait_weak
			}
			extra_trait_points = 3
		}
		create_country = {
			name = "The Dark Ones"
			name_list = "ART2"
			type = mem_phase_2_antaran_empire
			ignore_initial_colony_error = yes
			authority = auth_democratic
			civics = {
				civic = civic_warrior_culture
				civic = civic_nationalistic_zeal
			}
			species = last_created
			ethos = {
				ethic = ethic_xenophobe
				ethic = ethic_fanatic_militarist
			}
			flag = {
				icon = {
					category = "domination"
					file = "domination_21.dds"
				}
				background= {
					category = "backgrounds"
					file = "00_solid.dds"
				}
				colors={
					"black"
					"dark_purple"
					"null"
					"null"
				}
			}
			effect = {
				set_country_flag = mem_antaran_country	
				add_minerals = 5000
				add_energy = 5000		
				# must initialize global designs here
				add_global_ship_design = "Freki"
				add_global_ship_design = "Geri"
				add_global_ship_design = "Fenrir"
				add_global_ship_design = "Sleipnir"
				add_global_ship_design = "Midgard"		
			}
		}
		last_created_country = {
			set_policy = {
				policy = artificial_intelligence_policy
				option = ai_full_rights
				cooldown = yes
			}
			set_graphical_culture = mem_ancient_01
			save_global_event_target_as = mem_antaran_country
			add_minerals = 20000
			add_energy = 20000
		}
	}
}
